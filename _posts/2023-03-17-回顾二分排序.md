```javascript

let datas = [1, 0, 3, 2, 6, 4, 5];
if (datas.length > 0) {

    let hl = datas.length;
    for (let j = 1; j < hl; j++) {
        let h = datas[j];

        let start = 0;
        let end = j;
        let middle = 0;
        console.log("value:" + h);
        while (start <= end) {
            middle = Math.floor((start + end) / 2);
            console.log("-");
            console.log(start);
            console.log(end);
            console.log(middle);

            console.log("-");
            let v = datas[middle];
            if (v < h) {
              //调整偏移量
                middle = start = middle + 1;
            } else if (v > h) {
                end = middle - 1;
            } else {
                break;
            }
        }
        console.log("---------");
        console.log(start);
        console.log(end);
        console.log(middle);
        console.log(datas);
        console.log("---------");

        if (middle != j) {
            let buf = datas.splice(j, 1);
            datas.splice(middle, 0, buf[0]);
        }
        console.log(">>>>");
        console.log(datas);
        console.log("<<<<");
    }
    console.log(datas);
}
```
